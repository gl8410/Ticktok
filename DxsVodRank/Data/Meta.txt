命名方式：年-月-日_类型（4护肤，5彩妆）
"Vname":视频名称(string)
"UpName":主播名称(string)
"Vlong"：视频长度(num)
"Source"：商品来源(string)
"Uptime"：上传时间(datetime)
"UpID"：主播ID(num)
"VID"：视频ID(num)
"RelatedVs"：相关视频数量(num)
"newDayviews"：日新增浏览量(num)
"Sales"：销量(num)
"Price"：价格(num)
"Commission"：佣金(num)
"TotalLikes"：总点赞数(num)
"TotalComments"：总评论数(num)
"Changerate"：转化率(%)
"PLink"：商品链接(string)
